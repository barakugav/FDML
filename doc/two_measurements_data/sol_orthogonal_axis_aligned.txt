+ 2*d2^2*k^2*x^2
- d2^4*x^4
- d1^4*y^4
- k^4
+ 2*d1^2*k^2*y^2
- 4*k^2*x^2*y^2
+ 2*d1^2*d2^2*x^2*y^2
--------

A1 = A2 = d1^2
B1 = B2 = d2^2
D1 = D2 = -k^2

A1 * D2 + A2 * D1 = - 2 * d1^2 * k^2
B1 * D2 + B2 * D1 = - 2 * d2^2 * k^2
C1 * D2 + C2 * D1 = 0 => C1 = - C2
A1 * B2 + A2 * B1  + C1 * C2 = + 4*k^2 - 2*d1^2*d2^2
                             = 2*d1^2*d2^2 + C1 * C2

C1 * C2 = 4*k^2 - 4*d1^2*d2^2 = 2 * (d1^2*d2^2 - k^2)

C1 = C2 = 2 * sqrt(d1^2*d2^2 - k^2)
